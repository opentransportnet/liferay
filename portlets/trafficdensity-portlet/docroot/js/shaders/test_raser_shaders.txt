<script id="raster_fShader" type="x-shader/x-fragment">
      precision mediump float;  
 	  varying vec4 fCol;
	
	   
 

      void main() {
		vec4 col = vec4(0.5,0.5,0,1);	
		
		gl_FragColor = fCol;
      }
    </script>
    
    <script id="raster_vShader" type="x-shader/x-vertex">

      attribute vec4 wPoint;     
      
      uniform mat4 mapMatrix;
      uniform mat4 rasterMatrix;
   	  uniform sampler2D uSampler;
      
      varying vec4 fCol;   
     
       
      void main() {
		
  		float p_size = 1.;  	    	     	
		gl_PointSize = p_size;
		vec4 glp = mapMatrix*wPoint;
		
	    vec4 rp = rasterMatrix*glp;
		vec4 fdata = texture2D(uSampler, vec2(rp[0],rp[1]));
		
		gl_Position =  glp;
		fCol =fdata; // vec4(1.,0.,0.,1.);
			
 		
      }
    </script>
